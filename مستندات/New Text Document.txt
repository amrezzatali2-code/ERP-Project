  @* مجموعة أزرار التصدير على اليمين *@
  
  <div class="ms-auto d-flex flex-wrap gap-2">
      <form method="get"
            asp-action="Export"
            asp-controller="StockLedger"
            class="d-flex flex-wrap gap-2 align-items-center">

          @* الحفاظ على نفس الفلاتر الحالية وقت التصدير *@
          <input type="hidden" name="search" value="@search" />
          <input type="hidden" name="searchBy" value="@searchBy" />
          <input type="hidden" name="sort" value="@sort" />
          <input type="hidden" name="dir" value="@dir" />
          <input type="hidden" name="useDateRange" value="@(useDateRange ? "true" : "false")" />
          @if (fromDate.HasValue)
          {
              <input type="hidden" name="fromDate" value="@fromDate.Value.ToString("o")" />
          }
          @if (toDate.HasValue)
          {
              <input type="hidden" name="toDate" value="@toDate.Value.ToString("o")" />
          }

          <label class="form-label m-0">نوع التصدير</label>
          <select name="format" class="form-select form-select-sm">
              <option value="excel" selected>Excel</option>
              <option value="csv">CSV</option>
          </select>

          <button type="submit" class="btn btn-sm btn-outline-success">
              تصدير
          </button>
      </form>
  </div>
